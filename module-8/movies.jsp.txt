<%
    // Insert movie if form submitted
    String title = request.getParameter("title");
    String genre = request.getParameter("genre");
    String director = request.getParameter("director");
    String yearStr = request.getParameter("year");
    String ratingStr = request.getParameter("rating");

    if(title != null && genre != null && director != null && yearStr != null && ratingStr != null) {
        try {
            int year = Integer.parseInt(yearStr);
            double rating = Double.parseDouble(ratingStr);

            MovieBean movie = new MovieBean();
            movie.setTitle(title);
            movie.setGenre(genre);
            movie.setDirector(director);
            movie.setYear(year);
            movie.setRating(rating);

            boolean success = movie.insertMovie();
            if(success) {
%>
                <p style="color:green;">Movie added successfully!</p>
<%
            } else {
%>
                <p style="color:red;">Error adding movie.</p>
<%
            }

        } catch(Exception e) {
%>
            <p style="color:red;">Invalid input: <%= e.getMessage() %></p>
<%
        }
    }
%>
